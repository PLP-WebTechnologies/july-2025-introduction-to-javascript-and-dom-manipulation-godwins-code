// This function checks if a person is old enough to vote
function checkAge() {
  // Get the age value from the input field
  let age = parseInt(document.getElementById('ageInput').value);
  let resultElement = document.getElementById('ageResult');

  // Check if the input is valid
  if (isNaN(age)) {
    resultElement.textContent = "Please enter a valid number!";
    resultElement.style.color = "red";
  } else if (age >= 18) {
    resultElement.textContent = "You are eligible to vote!";
    resultElement.style.color = "green";
  } else {
    resultElement.textContent = "You are not yet eligible to vote.";
    resultElement.style.color = "orange";
  }
}

// Function to generate multiplication table using a for loop
function generateTable() {
  let number = parseInt(document.getElementById('tableNumber').value);
  let tableElement = document.getElementById('multiplicationTable');

  if (isNaN(number)) {
    tableElement.textContent = "Please enter a valid number!";
    return;
  }

  // Create the table using a for loop
  let tableHTML = "Multiplication Table:<br>";
  for (let i = 1; i <= 10; i++) {
    tableHTML += `${number} √ó ${i} = ${number * i}<br>`;
  }

  tableElement.innerHTML = tableHTML;
}

// Function to start a countdown using a while loop
function startCountdown() {
  let count = 10;
  let countdownElement = document.getElementById('countdown');

  // Use setInterval to create a timer
  let countdownInterval = setInterval(() => {
    countdownElement.textContent = `Countdown: ${count}`;
    if (count === 0) {
      clearInterval(countdownInterval);
      countdownElement.textContent = "Countdown: Blast off! üöÄ";
    } else {
      count--;
    }
  }, 1000);
}

// Function to add item to dynamic list
function addItem() {
  let input = document.getElementById('newItemInput');
  let list = document.getElementById('dynamicList');

  if (!input.value) {
    alert("Please enter an item!");
    return;
  }

  // Create a new list item element
  let newItem = document.createElement('div');
  newItem.className = 'todo-item';
  newItem.innerHTML = `<span class="todo-text">${input.value}</span><button class="todo-delete" onclick="this.parentElement.remove()">Delete</button>`;

  // Add the new item to the list
  list.appendChild(newItem);
  input.value = '';
}

// Function to change background color randomly
function changeColor() {
  let colors = ['#6c63ff', '#4e54c8', '#ff6b6b', '#47b881', '#f7a54a'];
  let colorBox = document.getElementById('colorBox');
  let randomColor = colors[Math.floor(Math.random() * colors.length)];
  colorBox.style.backgroundColor = randomColor;
  colorBox.textContent = `Color: ${randomColor}`;
}

// Function to toggle content visibility
function toggleContent() {
  let message = document.getElementById('secretMessage');
  if (message.style.display === 'none') {
    message.style.display = 'block';
  } else {
    message.style.display = 'none';
  }
}
